Authentication

Endpoint: /api/auth.php

Method: POST

Request Body:

{
    "email": "user@example.com",
    "password": "password123"
}

Response:

{
    "status": "success",
    "user_id": 123,
    "role": "client"
}

License Validation

Endpoint: /api/validate_license.php

Method: POST

Request Body:

{
    "license_key": "ABC123XYZ"
}

Response:

{
    "status": "valid",
    "message": "License is valid."
}

Orders

Endpoint: /api/orders.php

Method: POST

Request Body:

{
    "product_id": 1,
    "quantity": 2
}

Response:

{
    "status": "success",
    "message": "Order placed successfully."
}

Products

Endpoint: /api/products.php

Method: GET

Response:

{
    "status": "success",
    "products": [
        {"id": 1, "name": "Product A", "price": 19.99},
        {"id": 2, "name": "Product B", "price": 29.99}
    ]
}

Clients

Endpoint: /api/clients.php

Method: GET

Response:

{
    "status": "success",
    "clients": [
        {"id": 123, "name": "Client A", "email": "clientA@example.com"},
        {"id": 124, "name": "Client B", "email": "clientB@example.com"}
    ]
}
